## Zig concepten

### Overflow detectie — wrapping arithmetic

Normaal gesproken crasht Zig bij integer overflow (veiligheid). Maar in een emulator **wil** je overflow — de CPU doet dat expres.

```zig
// Dit crasht als het overloopt:
const result = a + b;  // NIET gebruiken voor emulatie!

// Dit wrapt rond zoals hardware doet:
const result, const carry = @addWithOverflow(a, b);
// result = a + b (gewrapt), carry = 1 als het overliep

// Of gewoon wrapping add:
const result = a +% b;  // +% = wrapping add, nooit crash
const result = a -% b;  // -% = wrapping sub
```

### De barrel shifter

De ARM CPU heeft een "barrel shifter" die een waarde kan shiften/roteren als onderdeel van een instructie. Er zijn 4 types:

```zig
fn barrel_shift(value: u32, shift_type: u2, amount: u5) struct { result: u32, carry: u1 } {
    return switch (shift_type) {
        0b00 => {  // LSL: Logical Shift Left — bits naar links, nullen rechts erin
            // 0b00001111 LSL 2 = 0b00111100
            const result = if (amount == 0) value else value << amount;
            // carry = het laatste bit dat "eruit viel"
        },
        0b01 => {  // LSR: Logical Shift Right — bits naar rechts, nullen links erin
            // 0b11110000 LSR 4 = 0b00001111
        },
        0b10 => {  // ASR: Arithmetic Shift Right — als LSR maar het teken-bit wordt herhaald
            // 0b10000100 ASR 2 = 0b11100001 (de 1 bovenaan wordt behouden)
            // Dit behoudt het teken bij signed getallen
        },
        0b11 => {  // ROR: Rotate Right — bits die rechts eruit vallen komen links terug
            // 0b00001111 ROR 4 = 0b11110000
        },
    };
}
```

### Signed vs unsigned in Zig

```zig
const unsigned: u32 = 0xFFFFFFFF;  // = 4294967295
const signed: i32 = @bitCast(unsigned);  // = -1 (zelfde bits, andere interpretatie)
```

`@bitCast` verandert het TYPE maar niet de BITS. Handig wanneer je signed arithmetic nodig hebt (bijv. bij SUB en CMP).

### De 16 opcodes

```
0x0 AND  Rd = Rn & Op2           logische AND
0x1 EOR  Rd = Rn ^ Op2           logische exclusive OR
0x2 SUB  Rd = Rn - Op2           aftrekken
0x3 RSB  Rd = Op2 - Rn           omgekeerd aftrekken
0x4 ADD  Rd = Rn + Op2           optellen
0x5 ADC  Rd = Rn + Op2 + Carry   optellen met carry
0x6 SBC  Rd = Rn - Op2 - !Carry  aftrekken met borrow
0x7 RSC  Rd = Op2 - Rn - !Carry  omgekeerd aftrekken met borrow
0x8 TST  flags van Rn & Op2      test bits (schrijft geen resultaat)
0x9 TEQ  flags van Rn ^ Op2      test gelijkheid (schrijft geen resultaat)
0xA CMP  flags van Rn - Op2      vergelijk (schrijft geen resultaat)
0xB CMN  flags van Rn + Op2      vergelijk negatief (schrijft geen resultaat)
0xC ORR  Rd = Rn | Op2           logische OR
0xD MOV  Rd = Op2                verplaats (Rn niet gebruikt)
0xE BIC  Rd = Rn & ~Op2          bit clear (wis bits)
0xF MVN  Rd = ~Op2               verplaats NOT (inversie)
```

### Flags berekenen

```zig
fn set_nz_flags(cpsr: *PSR, result: u32) void {
    cpsr.n = @truncate(result >> 31);  // bit 31 = teken-bit
    cpsr.z = if (result == 0) 1 else 0;
}

// Voor ADD: carry en overflow
fn add_flags(a: u32, b: u32) struct { result: u32, c: u1, v: u1 } {
    const result, const carry = @addWithOverflow(a, b);
    // Overflow bij signed: als beide operanden hetzelfde teken hebben
    // maar het resultaat een ander teken heeft
    const v: u1 = @truncate((~(a ^ b) & (a ^ result)) >> 31);
    return .{ .result = result, .c = carry, .v = v };
}
```

### Bitwise NOT (`~`)

```zig
const x: u32 = 0b00001111;
const y = ~x;
// y = 0b11111111111111111111111111110000  (alle bits omgedraaid)
```

Gebruikt bij BIC (`Rd = Rn & ~Op2`) en MVN (`Rd = ~Op2`).

### XOR (`^`) — exclusieve OR

```zig
  0b11001100
^ 0b10101010
= 0b01100110   // 1 alleen als de bits VERSCHILLEND zijn
```

Gebruikt bij EOR en TEQ.

## Stap-voor-stap

### Stap 1: `src/core/arm.zig` — Barrel shifter implementeren

De barrel shifter is het hart van ARM data processing. Hij kan een waarde shiften of roteren als onderdeel van ELKE ALU-instructie, zonder extra clock cycles.

**Waarom heeft ARM een barrel shifter in elke instructie?** Dit is een fundamentele ARM design-keuze die het onderscheidt van andere architecturen (x86, MIPS). Door de shift in de instructie zelf te verwerken, kun je in één instructie doen wat anders twee instructies zou kosten. Bijvoorbeeld `ADD R0, R1, R2, LSL #3` telt R1 op bij (R2 × 8) — in één klok-cyclus.

Er zijn 4 shift-types. Elk gedraagt zich anders bij een amount van 0:

```zig
const ShiftResult = struct { result: u32, carry: u1 };

fn barrelShift(value: u32, shift_type: u2, amount: u8, old_carry: u1) ShiftResult {
    // Amount 0 is speciaal bij sommige shift types
    if (amount == 0) {
        return switch (shift_type) {
            0b00 => .{ .result = value, .carry = old_carry },  // LSL #0 = geen shift
            0b01 => .{ .result = 0, .carry = @truncate(value >> 31) },  // LSR #0 → LSR #32
            0b10 => {  // ASR #0 → ASR #32
                const sign: u1 = @truncate(value >> 31);
                return .{
                    .result = if (sign == 1) 0xFFFFFFFF else 0,
                    .carry = sign,
                };
            },
            0b11 => {  // ROR #0 → RRX (rotate right extended through carry)
                return .{
                    .result = (@as(u32, old_carry) << 31) | (value >> 1),
                    .carry = @truncate(value),
                };
            },
        };
    }

    return switch (shift_type) {
        0b00 => {  // LSL: Logical Shift Left
            if (amount >= 32) return .{ .result = 0, .carry = if (amount == 32) @truncate(value) else 0 };
            const a: u5 = @intCast(amount);
            return .{
                .result = value << a,
                .carry = @truncate(value >> (@as(u5, 32 - a))),
            };
        },
        0b01 => {  // LSR: Logical Shift Right
            if (amount >= 32) return .{ .result = 0, .carry = if (amount == 32) @truncate(value >> 31) else 0 };
            const a: u5 = @intCast(amount);
            return .{
                .result = value >> a,
                .carry = @truncate(value >> (a - 1)),
            };
        },
        0b10 => {  // ASR: Arithmetic Shift Right — behoudt teken-bit
            if (amount >= 32) {
                const sign: u1 = @truncate(value >> 31);
                return .{ .result = if (sign == 1) 0xFFFFFFFF else 0, .carry = sign };
            }
            const a: u5 = @intCast(amount);
            const signed: i32 = @bitCast(value);
            return .{
                .result = @bitCast(signed >> a),
                .carry = @truncate(value >> (a - 1)),
            };
        },
        0b11 => {  // ROR: Rotate Right
            const rot: u5 = @truncate(amount);  // effectief amount mod 32
            const result = std.math.rotr(u32, value, @as(u32, rot));
            return .{
                .result = result,
                .carry = @truncate(result >> 31),
            };
        },
    };
}
```

**Waarom is amount 0 speciaal?** ARM hergebruikt "amount 0" voor speciale bewerkingen. `LSR #0` in de instructie-encoding wordt geïnterpreteerd als `LSR #32` (shift alle bits eruit). `ROR #0` wordt `RRX` — een 33-bit rotate die de carry flag erbij betrekt. Dit is gedocumenteerd in de ARM Architecture Reference Manual en games kunnen erop rekenen.

**Waarom ASR `@bitCast` naar i32 gebruikt:** ASR (Arithmetic Shift Right) moet het teken-bit (bit 31) behouden. In Zig doet `>>` op een `i32` automatisch een arithmetic shift (teken-bit wordt herhaald), terwijl `>>` op een `u32` een logical shift doet (nullen worden ingevuld). Door te casten naar signed, krijg je het juiste gedrag gratis.

**Valkuil:** De carry-output van de barrel shifter is belangrijk! Bij logische operaties (AND, ORR, EOR, etc.) met de S-bit aan, wordt de C-flag gezet op de barrel shifter carry, niet op de ALU carry. Dit is een subtiel maar belangrijk verschil.

### Stap 2: `src/core/arm.zig` — Operand2 decoderen

ARM data processing instructies hebben twee soorten "operand 2":

**Waarom twee vormen?** Dit geeft de programmeur flexibiliteit. Immediate is handig voor constanten (`ADD R0, R0, #1`), register met shift voor complexe berekeningen (`ADD R0, R1, R2, LSL R3`).

```zig
fn decodeOperand2(cpu: *const Cpu, instruction: u32) ShiftResult {
    const is_immediate = (instruction >> 25) & 1 == 1;  // bit 25 = I flag

    if (is_immediate) {
        // Immediate: 8-bit waarde geroteerd over even aantal posities
        const imm: u8 = @truncate(instruction);           // bits [7:0]
        const rotate: u4 = @truncate(instruction >> 8);    // bits [11:8]
        const rot_amount: u5 = @as(u5, rotate) * 2;       // altijd even

        if (rot_amount == 0) {
            return .{ .result = imm, .carry = cpu.cpsr.c };
        }

        const result = std.math.rotr(u32, @as(u32, imm), @as(u32, rot_amount));
        return .{ .result = result, .carry = @truncate(result >> 31) };
    } else {
        // Register: Rm (optioneel geshifted)
        const rm: u4 = @truncate(instruction);          // bits [3:0]
        const shift_type: u2 = @truncate(instruction >> 5);  // bits [6:5]
        const rm_value = cpu.get_reg(rm);

        const shift_by_reg = (instruction >> 4) & 1 == 1;  // bit 4
        if (shift_by_reg) {
            // Shift amount uit register Rs (bits [11:8])
            const rs: u4 = @truncate(instruction >> 8);
            const amount: u8 = @truncate(cpu.get_reg(rs));
            return barrelShift(rm_value, shift_type, amount, cpu.cpsr.c);
        } else {
            // Shift amount is een 5-bit immediate (bits [11:7])
            const amount: u5 = @truncate(instruction >> 7);
            return barrelShift(rm_value, shift_type, amount, cpu.cpsr.c);
        }
    }
}
```

**Waarom rotatie over een even aantal posities bij immediate?** ARM wil zoveel mogelijk waarden representeerbaar maken in 12 bits (4 bits rotatie + 8 bits waarde). Door alleen over even posities te roteren (0, 2, 4, ... 30), kun je veel voorkomende constanten coderen: 0xFF, 0xFF00, 0xFF0000, etc. Oneven rotaties zijn niet nodig omdat de meeste nuttige constanten al bereikbaar zijn.

**Valkuil bij register-shifted:** Als de shift amount uit een register komt (bit 4 = 1), wordt alleen de onderste byte van dat register gebruikt. Een shift amount van 33 is dus anders dan 1! En als Rm of Rs register 15 (PC) is, is de waarde PC+12 in plaats van PC+8 vanwege een extra fetch-cyclus.

### Stap 3: `src/core/arm.zig` — De 16 ALU operaties

Nu de kern: de 16 opcodes implementeren. De opcode zit in bits [24:21].

**Waarom wrapping arithmetic (+%, -%)?** Zig's standaard `+` en `-` paniekten bij overflow. Maar een CPU *moet* wrappen: `0xFFFFFFFF + 1 = 0x00000000` met carry. De `+%` operator doet exact dit — het is hetzelfde als wat de hardware doet.

```zig
fn dataProcessing(cpu: *Cpu, bus: *Bus, instruction: u32) void {
    _ = bus;
    const opcode: u4 = @truncate(instruction >> 21);  // bits [24:21]
    const s_bit = (instruction >> 20) & 1 == 1;        // bit 20: set flags?
    const rn: u4 = @truncate(instruction >> 16);        // bits [19:16]
    const rd: u4 = @truncate(instruction >> 12);        // bits [15:12]

    const op1 = cpu.get_reg(rn);
    const op2_result = decodeOperand2(cpu, instruction);
    const op2 = op2_result.result;
    const shifter_carry = op2_result.carry;

    // Voer de operatie uit
    var result: u32 = undefined;
    var carry: u1 = cpu.cpsr.c;
    var overflow: u1 = cpu.cpsr.v;
    var write_result = true;  // TST/TEQ/CMP/CMN schrijven geen resultaat

    switch (opcode) {
        0x0 => { result = op1 & op2; carry = shifter_carry; },                    // AND
        0x1 => { result = op1 ^ op2; carry = shifter_carry; },                    // EOR
        0x2 => { const r = subFlags(op1, op2); result = r.result; carry = r.carry; overflow = r.overflow; },  // SUB
        0x3 => { const r = subFlags(op2, op1); result = r.result; carry = r.carry; overflow = r.overflow; },  // RSB
        0x4 => { const r = addFlags(op1, op2); result = r.result; carry = r.carry; overflow = r.overflow; },  // ADD
        0x5 => {  // ADC: add with carry
            const r1 = addFlags(op1, op2);
            const r2 = addFlags(r1.result, @as(u32, cpu.cpsr.c));
            result = r2.result;
            carry = r1.carry | r2.carry;
            overflow = r1.overflow | r2.overflow;
        },
        0x6 => {  // SBC: subtract with carry
            const r1 = subFlags(op1, op2);
            const borrow: u32 = @as(u32, 1) - @as(u32, cpu.cpsr.c);
            const r2 = subFlags(r1.result, borrow);
            result = r2.result;
            carry = r1.carry & r2.carry;
            overflow = r1.overflow | r2.overflow;
        },
        0x7 => {  // RSC: reverse subtract with carry
            const r1 = subFlags(op2, op1);
            const borrow: u32 = @as(u32, 1) - @as(u32, cpu.cpsr.c);
            const r2 = subFlags(r1.result, borrow);
            result = r2.result;
            carry = r1.carry & r2.carry;
            overflow = r1.overflow | r2.overflow;
        },
        0x8 => { result = op1 & op2; carry = shifter_carry; write_result = false; },  // TST
        0x9 => { result = op1 ^ op2; carry = shifter_carry; write_result = false; },  // TEQ
        0xA => { const r = subFlags(op1, op2); result = r.result; carry = r.carry; overflow = r.overflow; write_result = false; },  // CMP
        0xB => { const r = addFlags(op1, op2); result = r.result; carry = r.carry; overflow = r.overflow; write_result = false; },  // CMN
        0xC => { result = op1 | op2; carry = shifter_carry; },                    // ORR
        0xD => { result = op2; carry = shifter_carry; },                          // MOV
        0xE => { result = op1 & ~op2; carry = shifter_carry; },                   // BIC
        0xF => { result = ~op2; carry = shifter_carry; },                         // MVN
    }

    // Schrijf resultaat (niet voor TST/TEQ/CMP/CMN)
    if (write_result) {
        cpu.set_reg(rd, result);
    }

    // Update flags als S-bit aan staat
    if (s_bit) {
        if (rd == 15) {
            // Speciale case: S-bit + Rd=PC → CPSR = SPSR (return from exception)
            cpu.cpsr = cpu.getSpsr();
        } else {
            cpu.cpsr.n = @truncate(result >> 31);
            cpu.cpsr.z = if (result == 0) @as(u1, 1) else @as(u1, 0);
            cpu.cpsr.c = carry;
            // Overflow alleen bij rekenkundige operaties (niet bij logische)
            if (opcode >= 0x2 and opcode <= 0x7 or opcode == 0xA or opcode == 0xB) {
                cpu.cpsr.v = overflow;
            }
        }
    }
}
```

**Waarom de S-bit apart checken?** Bit 20 (de S-bit) bepaalt of de instructie de CPSR flags update. Veel instructies willen berekenen *zonder* flags te veranderen (bijv. een adresberekening met ADD). Alleen als de S-bit aan staat, worden N, Z, C, V bijgewerkt. Dit geeft de programmeur fijnkorrelige controle.

**Waarom TST/TEQ/CMP/CMN geen resultaat schrijven:** Deze operaties zetten *alleen* flags. Ze berekenen wel een resultaat (voor de flags), maar schrijven het niet naar een register. De S-bit staat bij deze altijd impliciet aan (de encoder garandeert dit). Bits [15:12] (normaal Rd) worden bij deze operaties gebruikt voor andere doeleinden.

**Waarom het speciale geval Rd=15 + S-bit?** Als je resultaat naar PC schrijft MET de S-bit aan, kopieert de hardware SPSR → CPSR. Dit is het mechanisme om terug te keren van een exception: je herstelt zowel het adres (PC) als de processor state (CPSR) in één instructie. Bijvoorbeeld `MOVS PC, LR` in een IRQ handler.

### Stap 4: `src/core/arm.zig` — Flag helper functies

**Waarom aparte functies voor flags?** De flag-berekening is subtiel en foutgevoelig. Door het in helper functies te zetten, kun je ze apart testen en hergebruiken bij ADD, ADC, SUB, SBC, CMP, CMN.

```zig
const FlagResult = struct { result: u32, carry: u1, overflow: u1 };

fn addFlags(a: u32, b: u32) FlagResult {
    const result, const carry = @addWithOverflow(a, b);
    // Signed overflow: als de operanden hetzelfde teken hebben
    // maar het resultaat een ander teken heeft
    const v: u1 = @truncate((~(a ^ b) & (a ^ result)) >> 31);
    return .{ .result = result, .carry = carry, .overflow = v };
}

fn subFlags(a: u32, b: u32) FlagResult {
    const result, const borrow = @subWithOverflow(a, b);
    // Bij ARM SUB: carry = GEEN borrow (omgekeerd!)
    const carry: u1 = ~borrow;
    // Signed overflow: operanden verschillend teken, resultaat ander teken dan a
    const v: u1 = @truncate(((a ^ b) & (a ^ result)) >> 31);
    return .{ .result = result, .carry = carry, .overflow = v };
}
```

**Waarom is carry bij SUB omgekeerd?** In ARM is de carry flag bij aftrekking gedefinieerd als "geen borrow". Dus `5 - 3` geeft carry=1 (geen borrow nodig), maar `3 - 5` geeft carry=0 (er was een borrow). Dit is het tegenovergestelde van wat `@subWithOverflow` retourneert, vandaar de `~borrow`.

**Waarom die XOR-formule voor overflow?** Signed overflow treedt op als:
- Bij ADD: beide operanden positief maar resultaat negatief, of beide negatief maar resultaat positief
- Bij SUB: `a` en `b` verschillend teken, en resultaat verschilt van `a`

De formule `(~(a ^ b) & (a ^ result)) >> 31` checkt precies dit in één expressie:
- `a ^ b`: bits waar het teken verschilt (voor ADD: we willen bits waar het teken GELIJK is → `~`)
- `a ^ result`: bits waar het resultaat een ander teken heeft dan `a`
- De `& >> 31` pakt alleen bit 31 (het teken-bit)

### Stap 5: `src/core/arm.zig` — Testen schrijven

Schrijf tests per opcode om correctheid te verifiëren:

```zig
test "MOV R0, #42" {
    var cpu = Cpu.init();
    var bus = Bus.init();
    // E3A00_02A = MOV R0, #42 (cond=AL, I=1, opcode=MOV, Rd=R0, imm=42)
    const instruction: u32 = 0xE3A0002A;
    dataProcessing(&cpu, &bus, instruction);
    try std.testing.expectEqual(@as(u32, 42), cpu.r[0]);
}

test "ADDS overflow detectie" {
    var cpu = Cpu.init();
    var bus = Bus.init();
    cpu.r[0] = 0x7FFFFFFF;  // max positief signed
    cpu.r[1] = 1;
    // E0900001 = ADDS R0, R0, R1 (S-bit aan)
    const instruction: u32 = 0xE0900001;
    dataProcessing(&cpu, &bus, instruction);
    try std.testing.expectEqual(@as(u32, 0x80000000), cpu.r[0]);
    try std.testing.expectEqual(@as(u1, 1), cpu.cpsr.n);  // negatief
    try std.testing.expectEqual(@as(u1, 0), cpu.cpsr.z);  // niet nul
    try std.testing.expectEqual(@as(u1, 0), cpu.cpsr.c);  // geen carry
    try std.testing.expectEqual(@as(u1, 1), cpu.cpsr.v);  // overflow!
}

test "CMP zet flags zonder register te schrijven" {
    var cpu = Cpu.init();
    var bus = Bus.init();
    cpu.r[0] = 42;
    cpu.r[1] = 42;
    // E1500001 = CMP R0, R1
    const instruction: u32 = 0xE1500001;
    dataProcessing(&cpu, &bus, instruction);
    try std.testing.expectEqual(@as(u1, 1), cpu.cpsr.z);  // 42 - 42 = 0, Z set
    try std.testing.expectEqual(@as(u1, 1), cpu.cpsr.c);  // geen borrow
}

test "barrel shifter LSL" {
    const result = barrelShift(0b1, 0b00, 4, 0);
    try std.testing.expectEqual(@as(u32, 0b10000), result.result);
}
```

**Tip:** Encodeer instructies met de hand is foutgevoelig. Maak een helper of gebruik constanten:
```zig
// Helper om een data processing instructie te bouwen
fn makeDataProc(cond: u4, i: u1, opcode: u4, s: u1, rn: u4, rd: u4, operand2: u12) u32 {
    return (@as(u32, cond) << 28) | (@as(u32, 0b00) << 26) |
           (@as(u32, i) << 25) | (@as(u32, opcode) << 21) |
           (@as(u32, s) << 20) | (@as(u32, rn) << 16) |
           (@as(u32, rd) << 12) | @as(u32, operand2);
}
```
